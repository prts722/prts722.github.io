# 定义获取源代码函数
def get_response(html_url):    # 函数参数(html_url)：要请求的链接（url）
    # 定义请求头
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36 Edg/122.0.0.0'
    }
    # 获取小说源代码
    html_data = requests.get(url=html_url, headers=headers)
    # 定义编码格式
    html_data.encoding = html_data.apparent_encoding
    # 返回源代码
    return html_data

通过上面代码的调用可以获得一个response=200的返回值，表示已经爬取成功，通过.text方法就可以查看页面源代码

# 定义获取各章节链接的函数
def get_data(html_url_2):   # 函数参数(html_url_2)： 小说目录页链接
    # 获取目录页源代码
    html_data = get_response(html_url_2).text
    # 提取小说各章节链接
    html = re.findall('<dd><a href ="(.*?)">', html_data)
    # 返回小说名字和章节链接
    return html

def search(word):
    s_url = f'https://www.c336.icu/user/search.html?q={word}&so=undefined'
    s_headers = {
	“请求标头”
}
 s_html = requests.get(url=s_url,headers=s_headers)
    cookie = s_html.cookies
    s_cookie = dict_from_cookiejar(cookie)
    s_html_2 = requests.get(url=s_url, cookies=s_cookie)
    get_json = json.loads(s_html_2.content)
    book_name_list = []
    book_url_list = []
    try:
        if get_json == 1:
            print("\r没有该小说或请稍后再试")
        else:
            book_url = re.findall('"url_list":"(.*?)",', s_html_2.text)
            book_name = re.findall('"articlename":"(.*?)",', s_html_2.text)
            # print(book_url)
            # print(book_name)
            for new_url in book_url:
                book_url_list.append(str(new_url).replace("\\","" ))
            for new_book_name in book_name:
                book_name_list.append(eval(repr(new_book_name).replace('\\\\', '\\')))

    except():
        print("请稍后再试")
    book_dict = dict(zip(book_name_list, book_url_list))
    return book_dict

上述方法能完美解决动态渲染问题，同时加入了报错解决以及查询不到小说的处理

def save(name, title, result):   # 三个参数含义：小说名，章节名，内容
    # 判断是否有小说文件夹，如果没有则自动创建
    file = '{}\\'.format(name)
    if not os.path.exists(file):
        os.mkdir(file)
    # 保存小说内容至文件夹
    with open(file + title + '.txt', mode='a', encoding='utf-8') as f:
        f.write(title)
        f.write('\n')
        f.write(result)
        f.write('\n')

# 搜索小说
x = str(input("请输入小说："))
n = quote(x,encoding="utf-8")
request_dict = search(str(n).replace(",","\n"))
print("您搜索到的小说结果如下")
# 选择小说
for name in request_dict:
    print(name)
choose_book = input("请选择要下载的小说：\n")
if choose_book in request_dict.keys():
    num = 1
    # 获得对应书籍链接
    book_href = "https://www.c336.icu" + request_dict[choose_book]
    # 爬取目录页，获取各章节链接
    book_text_href = get_data(book_href)
    # 通过for循环对每一个章节链接进行爬取并保存
    for text_href in book_text_href:
        new_href = "https://www.c336.icu" + text_href
        # print(new_href)
        text_name, text_result = content(new_href)
        str_text_name = str(*text_name)
        str_text_result = str(*text_result).replace("<br />","\n")
        print(f"正在保存第{num}节")
        save(choose_book, str_text_name, str_text_result)
        num += 1
        time.sleep(3)

else:
    print("没有这本小说")

import requests
from bs4 import BeautifulSoup

# 定义请求头，把程序伪装成浏览器请求，从而实现请求不被阻拦
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0.0.0 Safari/537.36'
}
for start_num in range(0, 250, 25):
    # 用get方法传入想爬取的链接
    response = requests.get('https://movie.douban.com/top250?start={}'.format(start_num), headers=headers)
    html = response.text   # 获取源代码(不加text获得的是访问回应值)
    soup = BeautifulSoup(html, 'html.parser')  # 用bs解析源代码
    all_title = soup.findAll('span', attrs={'class': 'title'})  # 查找所需内容
    for title in all_title:
        title_string = title.string
        if '/' not in title_string:
            print(title.string)

import requests
import re

# 定义请求头
headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0.0.0 Safari/537.36 Edg/121.0.0.0'
    }

# for 循环遍历每一个内容介绍的链接（也可以使用嵌套循环）
for i in range(1, 96):
    url = 'https://ssr1.scrape.center/detail/{}'.format(i)   # 需要爬取的链接
    cinema_1 = requests.get(url=url, headers=headers)    # 抓取信息（源代码）
    cinema = cinema_1.text  # 转成源代码
    title = re.findall('>(.*?)</a>', cinema)[0].replace('</h2', '')    # 抓取标题
    introduction = re.findall('<p data-v-63864230="">(.*?)<', cinema, re.S)[0].strip().replace('。', '\n')    # 抓取内容

# 创建txt保存抓取内容
    with open('电影简介' + '.txt', mode='a', encoding=cinema_1.apparent_encoding)as f:
        f.write(str(title))    # 写入电影名
        f.write('\n')          # 空行
        f.write(str(introduction))    # 写入简介内容
        f.write('\n')          # 空行
        f.write('\n')
        # 打印标题
    print(title + '保存成功')
